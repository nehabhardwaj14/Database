TABLE COMMANDS

CREATE TABLE

SYNTAX:

        CREATE TABLE TABLE_NAME(Column1 datatype (size) constraints,
        column2 datatype (size) constraint,....);
        
EXAMPLE:

        mysql> create table student(ROLL_NO int(2),NAME varchar(20),
        class varchar(20),ADDRESS varchar(30));
        Query OK, 0 rows affected, 1 warning (0.05 sec)
