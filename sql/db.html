DATABASE COMMANDS

CREATE DATABASE

        
SYNTAX:

        CREATE DATABASE DATABASE_NAME;
        
Example:

              mysql> create database class;
              Query OK, 1 row affected (0.03 sec)

SHOW DATABASE

        
SYNTAX:

         SHOW DATABASES;
        
EXAMPLE:

        mysql> show databases;
        +--------------------+
        | Database           |
        +--------------------+
        | class              |
        | collage            |
        | information_schema |
        | mysql              |
        | performance_schema |
        | practice           |
        | sys                |
        +--------------------+
        7 rows in set (0.04 sec)

USE DATABASE

        
SYNTAX:

          USE DATABASE_NAME;
        
EXAMPLE:

        mysql> use class;
        Database changed

DROP DATABASE

        
SYNTAX:

         DROP DATABASE DATABASE_NAME;
         
EXAMPLE:

         mysql> drop database class;
         Query OK, 1 row affected (0.06 sec)

         mysql> show databases;
         +--------------------+
         | Database           |
         +--------------------+
         | collage            |
         | information_schema |
         | mysql              |
         | performance_schema |
         | practice           |
         | sys                |
         +--------------------+
         6 rows in set (0.00 sec)
