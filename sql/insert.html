INSERT COMMAND

       
SYNTAX:


       1.   INSET INTO TABLE_NAME VALUES(value1,value2,value3,...);

       2.   INSET INTO TABLE_NAME(column1,column2,...)  VALUES(value1,value2,...);

       3.   INSET INTO TABLE_NAME VALUES(value1,value2,value3),
            (value1,value2,value3),
            (value1,value2,value3);

EXAMPLE:

       mysql> insert into student values(1,"Neha","BCA","A",98,24000,20);
                           OR
       mysql> insert into student values(2,"Abhishek","BCA","B", 90,24000,19),

       mysql>(3,"Asha","BCA","A" ,80,24000,19),(4,"Riya","BCA","B", 80,24000,20),

       mysql>(5,"Priya","BCA","A",30,24000 ,18), (6,"Misbah","BCA",",B",79,24000,19),

       mysql>(7,"Anil","BCA","A",10,24000,20),(8,"Aman","BCA","B",79,24000,19), 

       mysql> (9,"Pawan","BCA","A",75,24000,19);
